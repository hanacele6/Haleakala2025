ファイルの配置はこんな感じだといいです。

Mercury/                         # 全体の大元のフォルダ
│
├── data/                        # ① 生データフォルダ
│   ├── 20250811/                # 観測日ごとのフォルダ（YYYYMMDD）
│   │   ├── me01.fits            # 水星の生画像
│   │   └── sky01.fits           # 空の生画像 など
│   └── 20250813/
│
└── Haleakala2025/               # ② プロジェクトフォルダ【コードを置くところ】
    │
    ├── config.yaml              # ★解析パラメータを調整する「設定ファイル」
    ├── main.py                  # パイプライン全体を動かすコード
    ├── step00_... ～ step14_... # 各処理のプログラム部品
    │
    ├── SolarSpectrum.txt        # 太陽の参照スペクトルデータ（Step10で使用）
    ├── PA_correction2.txt       # 位相角の補正用データ（Step14で使用）
    │
    ├── kernels/                 # SPICE（天体の軌道計算ツール）用の辞書データ
    │   ├── lsk/
    │   ├── pck/
    │   └── spk/
    │
    ├── 2026ver/csvs/            # 解析に必要なメタデータが自動生成されるCSV
    │   ├── mcparams20250811.csv
    │   └── mcparams20250813.csv
    │
    └── output/                  # ③ 出力フォルダ
        │
        ├── 20250811/            # パイプラインを実行すると自動で作成・整理される
        │   ├── 1_fits/          # ノイズ除去や波長較正が終わったキレイな2次元画像
        │   ├── 2_spectra/       # 抽出された1次元のスペクトルデータ (.dat, .exos)
        │   └── 3_plots/         # 解析が上手くいったか確認するためのグラフ画像 (.png)
        │
        ├── All_Results_Summary_1D_2025.csv      # 1年分の最終データが結合されたCSV
        └── Correction_Factor_vs_TAA_1D_2025.png # TAA（真近点角）の最終プロット画像
